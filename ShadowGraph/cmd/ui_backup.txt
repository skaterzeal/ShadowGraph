var uiCmd = &cobra.Command{
	Use:   "ui",
	Short: "İnteraktif Zafiyet Haritası (Node Graph) panelini başlatır",
	Run: func(cmd *cobra.Command, args []string) {
		fmt.Printf("[\033[36m👉\033[0m] Büyüleyici Görsel Panel %d portunda başlatıldı!\n", port)
		fmt.Printf("[\033[32m👉\033[0m] Tarayıcı üzerinden gidin: http://localhost:%d\n\n", port)
		fmt.Println("[\033[33m!\033[0m] Panelden çıkmak için CTRL+C yapabilirsiniz.")
		
		http.HandleFunc("/api/graph", func(w http.ResponseWriter, r *http.Request) {
			data, err := db.GetGraphData()
			if err != nil {
				http.Error(w, err.Error(), http.StatusInternalServerError)
				return
			}
			w.Header().Set("Content-Type", "application/json")
			w.Write(data)
		})

		http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
			w.Header().Set("Content-Type", "text/html")
			w.Write([]byte(htmlContent))
		})
		
		if err := http.ListenAndServe(fmt.Sprintf(":%d", port), nil); err != nil {
			fmt.Println("[-] Sunucu hatası:", err)
		}
	},
}

func init() {
	rootCmd.AddCommand(uiCmd)
	uiCmd.Flags().IntVarP(&port, "port", "p", 8080, "UI panelinin çalışacağı port numarası")
}
